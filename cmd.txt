### Final Nginx Configuration for /etc/nginx/sites-available/wetoo

# This server block redirects all HTTP traffic to HTTPS.
server {
    listen 80;
    server_name social.me2we2.com;
    return 301 https://$host$request_uri;
}

# This server block handles all HTTPS traffic.
server {
    listen 443 ssl http2;
    server_name social.me2we2.com;

    # --- SSL Configuration ---
    # IMPORTANT: Replace these paths with the actual paths to your SSL certificate files.
    # These are the standard paths provided by Certbot (Let's Encrypt).
    ssl_certificate /etc/letsencrypt/live/social.me2we2.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/social.me2we2.com/privkey.pem;
    # The following two lines are recommended for strong security. Ensure these files exist.
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;


    # --- Frontend Application --- 
    # Serves your compiled React/UmiJS application.
    location / {
        root /home/edited_we2/frontend/dist;
        try_files $uri /index.html;
    }

    # --- Backend API --- 
    # Forwards all requests starting with /api/ to your Gunicorn/Django application.
    location /api/ {
        proxy_pass http://127.0.0.1:8000/api/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # --- Django Static & Media Files ---
    # Serves files collected by Django's 'collectstatic' command.
    location /static-files/ {
        alias /home/edited_we2/backend/core/static/;
    }

    # Serves user-uploaded files (e.g., profile pictures).
    location /media-files/ {
        alias /home/edited_we2/backend/core/media/;
    }
}
